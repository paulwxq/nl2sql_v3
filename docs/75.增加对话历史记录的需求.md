## 需求：
我希望在项目中回答NL问题的时候，或者根据NL生成SQL的时候，增加用户的历史对话记录，这样在回答NL问题的时候，或者根据NL生成SQL的时候，可以参考用户的历史对话记录，从而给出更准确的回答或者SQL。
## 现状：
当前项目已经使用 langgraph-checkpoint-postgres 的api,把用户的对话历史记录，保存在了postgresql数据库的 langgraph.store表中。
## store 相关表的定义信息如下：
create table langgraph.store
(
    prefix      text  not null,
    key         text  not null,
    value       jsonb not null,
    created_at  timestamp with time zone default CURRENT_TIMESTAMP,
    updated_at  timestamp with time zone default CURRENT_TIMESTAMP,
    expires_at  timestamp with time zone,
    ttl_minutes integer,
    primary key (prefix, key)
);

alter table langgraph.store
    owner to postgres;

create index store_prefix_idx
    on langgraph.store (prefix text_pattern_ops);

create index idx_store_expires_at
    on langgraph.store (expires_at)
    where (expires_at IS NOT NULL);

create table langgraph.store_migrations
(
    v integer not null
        primary key
);

alter table langgraph.store_migrations
    owner to postgres;
